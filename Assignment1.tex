\section {Properties of asymptotic bounds}
\begin{question} (25 points) \label{q2}
Let $f(n)$ and $g(n)$ be asymptotically positive functions.

\begin{definition}
    We say that a function $f: \mathbb{N} \to \mathbb{N}$ is \textbf{unbounded increasing monotone function} if:
    \begin{enumerate}
        \item for all $x,y$:
        \[\text{\hspace{6cm}}x < y \iff f(x) < f(y) \text{\hspace{1.4cm} ($\iff$ stands for if and only if)}\]

        
        \item for all constant $M\in \mathbb{R}^+$ there is $n_0 \in \mathbb{N}$ such that for all $n > n_0$:
        \[f(n) > M\]
    \end{enumerate}
\end{definition}

\begin{boxxed}
    \textcolor{CB_blue}{
    The logarithm function is an \textbf{unbounded increasing monotone function} defined for $x > 0$.}
\end{boxxed}

\begin{boxxed}
    \textcolor{CB_blue}{
    In conjectures \ref{q2:log} and \ref{q2:exp} you can assume that for all constant $c > 0$, there is $x_0 > 0$ such that for all $x > x_0$: \[c < \frac{x}{\log x}\]}
\end{boxxed}

\vspace{1ex}
\begin{enumerate}
    \item (4 points) $n > \log n $ for all $n \in \mathbb{N}$
    \item (2 points) Reflexivity: $f(n) \in \Theta(f(n))$
    \item (4 points) Transitivity: If $f(n) \in \Theta(g(n))$ and $g(n) \in \Theta(h(n))$ then $f(n) \in \Theta(h(n))$
    \item (3 points) Transpose Symmetry: $f(n) \in O(g(n)) \iff g(n) \in \Omega(f(n))$
    \item (4 points) For all $f(n) > 1$ and $g(n) > 1$ two \textbf{\textcolor{Bittersweet}{unbounded increasing monotone functions}} such that $f(n) \in \Omega(g(n))$, imply \[\log (f(n)) \in \Omega(\log (g(n)))\]

    \item (4 points) \label{q2:log} For all \textbf{constant} $k \ge 1$ and for all \textbf{constant} $\varepsilon > 0$:
    \[ \log^k (n) \in O (n^\varepsilon) \]
    \item (4 points) \label{q2:exp} For all constant $k \in \mathbb{N}$ and for all constant $\varepsilon > 0$ :
    \[n^k \in O( (1 + \varepsilon)^n)\]
\end{enumerate}
\end{question}
\bigskip
\begin{answer}
\textbf {Question 1:}
\medskip
Proof by induction on $n$.
\noindent\textbf{Base case} ($n = 1$):
\[
    1 > \log_2(1) = 0 \quad \checkmark
\]

\noindent\textbf{Induction hypothesis:} Assume that for some $n \in \mathbb{N}$, we have $n > \log n$.
We will show that $n + 1 > \log(n + 1)$.

\medskip
\[ 
n+1 = \log_2({2^{n+1}}) > \log_2({n+1})
\]
\begin{center}
(By the sub-claim and the fact that $\log$ is a monotonically increasing function)
\end{center}

\noindent\textbf{Sub-claim:} $2^{n+1} > n + 1$ for all $n \in \mathbb{N}$, proven by induction.

\medskip
\noindent\underline{Base case of sub-claim} ($n = 1$):
\[
    2^{1+1} = 2^2 = 4 > 2 = 1 + 1 \quad \checkmark
\]

\noindent\underline{Inductive step of sub-claim:} Assume $2^{n+1} > n + 1$. We show $2^{n+2} > n + 2$:
\[
    2^{n+2} = 2^{n+1} \cdot 2
    > (n+1) \cdot 2
    = 2n + 2
    > n + 2
\]
since $2n > n \iff n > 0$, which holds for all $n \in \mathbb{N}$. $\checkmark$

\medskip
\noindent Therefore $2^{n+1} > n + 1$ for all $n \in \mathbb{N}$, which implies $n + 1 > \log(n+1)$,
completing the induction. $\checkmark$ \\
\medskip
\textbf {Question 2:}
\medskip
We prove $f(n) \in \Theta(f(n))$ by showing both $f(n) \in O(f(n))$ and $f(n) \in \Omega(f(n))$.

\medskip
\noindent\underline{Part 1: $f(n) \in O(f(n))$}

We need $c_1, n_0 \in \mathbb{N}$ such that $f(n) \leq c_1 \cdot f(n)$ for all $n > n_0$:
\[
    f(n) \leq c_1 \cdot f(n) \iff 1 \leq c_1
\]
Choose $c_1 = 1$,\ $n_0 = 1$ --- a valid choice since $1 \leq 1$. $\checkmark$

\medskip
\noindent\underline{Part 2: $f(n) \in \Omega(f(n))$}

We need $c_2, n_0 \in \mathbb{N}$ such that $f(n) \geq c_2 \cdot f(n)$ for all $n > n_0$:
\[
    f(n) \geq c_2 \cdot f(n) \iff 1 \geq c_2
\]
Choose $c_2 = 1$,\ $n_0 = 1$ --- a valid choice since $1 \geq 1$. $\checkmark$

\medskip
Since both conditions hold, $f(n) \in \Theta(f(n))$. $\checkmark$ \\
\medskip
\textbf {Question 3:} \\
\noindent If $f(n) \in \Theta(g(n))$ and $g(n) \in \Theta(h(n))$, then $f(n) \in \Theta(h(n))$.

\medskip
\noindent Given $f(n) \in \Theta(g(n))$ and $g(n) \in \Theta(h(n))$.
We will show $f(n) \in \Theta(h(n))$, i.e., we prove two parts:
\begin{enumerate}[(1)]
    \item $f(n) \in O(h(n))$
    \item $f(n) \in \Omega(h(n))$
\end{enumerate}

\medskip
\noindent\underline{Proof of (1): $f(n) \in O(h(n))$}

We need $c_i, n_0 \in \mathbb{N}$ such that $f(n) \leq c_i \cdot h(n)$ for all $n > n_0$.

Since $f(n) \in \Theta(g(n))$, there exist $c,\, n_0^{(1)}$ such that:
\[
    f(n) \leq c \cdot g(n) \quad \text{for all } n > n_0^{(1)}
\]
Since $g(n) \in \Theta(h(n))$, there exist $c',\, n_0^{(2)}$ such that:
\[
    g(n) \leq c' \cdot h(n) \quad \text{for all } n > n_0^{(2)}
\]
Therefore, for all $n > \max\!\left(n_0^{(1)},\, n_0^{(2)}\right)$:
\[
    f(n) \leq c \cdot g(n) \leq c \cdot c' \cdot h(n) = c_i \cdot h(n)
    \quad \checkmark
\]

\medskip
\noindent\underline{Proof of (2): $f(n) \in \Omega(h(n))$}

We need $c_i, n_0 \in \mathbb{N}$ such that $f(n) \geq c_i \cdot h(n)$ for all $n > n_0$.

Since $f(n) \in \Theta(g(n))$, there exist $c,\, n_0^{(1)}$ such that:
\[
    f(n) \geq c \cdot g(n) \quad \text{for all } n > n_0^{(1)}
\]
Since $g(n) \in \Theta(h(n))$, there exist $c',\, n_0^{(2)}$ such that:
\[
    g(n) \geq c' \cdot h(n) \quad \text{for all } n > n_0^{(2)}
\]
Therefore, for all $n > \max\!\left(n_0^{(1)},\, n_0^{(2)}\right)$:
\[
    f(n) \geq c \cdot g(n) \geq c \cdot c' \cdot h(n) = c_i \cdot h(n)
    \quad \checkmark
\]

\medskip
\noindent Both (1) and (2) hold, so $f(n) \in \Theta(h(n))$. $\checkmark$

\textbf{Question 4:}
\noindent $f(n) \in O(g(n)) \iff g(n) \in \Omega(f(n))$
 
\bigskip
\noindent\textbf{($\Rightarrow$)} Assume $f(n) \in O(g(n))$. We show $g(n) \in \Omega(f(n))$.
 
There exist $c_1,\, n_0 \in \mathbb{N}$ such that for all $n > n_0$:
\[f(n) \leq c_1 \cdot g(n)\] 
from $f(n) \in O(g(n))$ \\
Dividing both sides by $c_1$:
\[
    \frac{1}{c_1} \cdot f(n) \leq g(n)
    \quad\Longrightarrow\quad
    g(n) \geq \frac{1}{c_1} \cdot f(n)
\]
Setting $c_2 = \dfrac{1}{c_1}$, we get $g(n) \geq c_2 \cdot f(n)$ for all $n > n_0$,
so $g(n) \in \Omega(f(n))$. $\checkmark$
 
\bigskip
\noindent\textbf{($\Leftarrow$)} Assume $g(n) \in \Omega(f(n))$. We show $f(n) \in O(g(n))$.
 
There exist $c_4,\, n_0 \in \mathbb{N}$ such that for all $n > n_0$:
\[g(n) \geq c_4 \cdot f(n)\]
from $g(n) \in \Omega(f(n))$

Dividing both sides by $c_4$:
\[
    \frac{1}{c_4} \cdot g(n) \geq f(n)
    \quad\Longrightarrow\quad
    f(n) \leq \frac{1}{c_4} \cdot g(n)
\]
Setting $c_3 = \dfrac{1}{c_4}$, we get $f(n) \leq c_3 \cdot g(n)$ for all $n > n_0$, \\
so $f(n) \in O(g(n))$. $\checkmark$ \\

\bigskip

\textbf{Question 5:}
\noindent\underline{Proof: $\log(f(n)) \in \Omega(\log(g(n)))$}\\
Since $f(n) \in \Omega(g(n))$, there exist constants $n_0$ and $c > 0$ such that for all $n > n_0$:
\[
f(n) \ge c \cdot g(n)
\]
According to the monotonicity of the $\log$ function:
\[
\log(f(n)) \ge \log(c \cdot g(n)) = \log(c) + \log(g(n))
\]
\medskip
We now show $\log(c) + \log(g(n)) \geq \frac{1}{2}\log(g(n))$ for large enough $n$.
Since $g(n)$ is an unbounded increasing monotone function, $\log(g(n)) \to \infty$.
Therefore, there exists $n_1 > n_0$ such that for all $n > n_1$:
\[
\log(g(n)) \geq |\log(c)|
\]
This implies $\log(c) \geq -\log(g(n))$, and therefore:
\[
\log(c) + \log(g(n)) \geq -\log(g(n)) + \log(g(n)) = \log(g(n)) \cdot \frac{1}{2} \cdot 2 \geq \frac{1}{2}\log(g(n))
\]
Setting $c' = \frac{1}{2}$, we get $\log(f(n)) \geq c' \cdot \log(g(n))$ for all $n > n_1$,
so $\log(f(n)) \in \Omega(\log(g(n)))$. $\checkmark$ \\

\bigskip

\textbf{Question 6:}
\noindent\underline{Proof:}\\
Let $k \geq 1$ and $\varepsilon > 0$ be constants.
We substitute $n = 2^m$, so $\log n = m$ and $n^\varepsilon = 2^{m\varepsilon}$.
The claim becomes:
\[
m^k \leq c \cdot (2^\varepsilon)^m \quad \text{for some constant } c > 0 \text{ and all large } m
\]
We apply the given hint with constant $\dfrac{k}{\varepsilon \log 2}$:
there exists $m_0$ such that for all $m > m_0$:
\[
\frac{k}{\varepsilon \log 2} < \frac{m}{\log m}
\implies k \log m < \varepsilon m \log 2
\implies \log(m^k) < \log(2^{\varepsilon m})
\]
Since $\log$ is a monotone increasing function:
\[
m^k < 2^{\varepsilon m} = (2^\varepsilon)^m = n^\varepsilon
\]
Setting $c = 1$ and $n_0 = 2^{m_0}$, we get $\log^k(n) \leq c \cdot n^\varepsilon$ for all $n > n_0$,
so $\log^k(n) \in O(n^\varepsilon)$. $\checkmark$ \\

\bigskip

\textbf{Question 7:}
\noindent\underline{Proof:}\\
Let $k \in \mathbb{N}$ and $\varepsilon > 0$ be constants.
To prove $n^k \in O((1+\varepsilon)^n)$ we will find a constant $c$ and $n_0$ so that for all $n > n_0$, $n^k \leq c \cdot (1 + \varepsilon)^n$.\\
\medskip
We set $c = 1$. Since $\log$ is a monotone function, $n^k \leq (1+\varepsilon)^n$ is equivalent to:
\[
k\log n \leq n\log(1+\varepsilon)
\]
which rearranges to:
\[
\frac{k}{\log(1+\varepsilon)} \leq \frac{n}{\log n}
\]
Since $\varepsilon > 0$, we have $\log(1+\varepsilon) > 0$, so $\dfrac{k}{\log(1+\varepsilon)}$ is a positive constant.
By the given hint, there exists $n_0$ such that for all $n > n_0$:
\[
\frac{k}{\log(1+\varepsilon)} < \frac{n}{\log n} \quad \checkmark
\]
Reversing the steps, this implies $k\log n < n\log(1+\varepsilon)$, and since $\log$ is monotone increasing:
\[
n^k < (1+\varepsilon)^n = c \cdot (1+\varepsilon)^n
\]
Therefore $n^k \in O((1+\varepsilon)^n)$. $\checkmark$

\end{answer}

\section {Recurrence Examples}
\begin{question} (15 points)
Find and prove an asymptotic tight bound $\Theta$ for $T(n)$ in each of the following recurrences. Assume that $T(c) = c$ for $c \le 4$.
\begin{enumerate}
    \item $T(n) = 4T(\frac{n}{2}) + n$
    \item $T(n) = T(\frac{3}{7}n) + 1$
    \item $T(n) = 6T(\frac{n}{2}) + n^4 \log{n}$
    \item $T(n) = 8T(\frac{n}{2}) + n^4 + 3n^3  \log{n} $
    \item $T(n) = T(\sqrt{n}) + n $
\end{enumerate}
\end{question}

\begin{answer} 
We apply the \textbf{Master Theorem} with parameters:
\[
    a = 4, \quad b = 2, \quad f(n) = n, \quad
    n^{\log_b a} = n^{\log_2 4} = n^2
\]

\noindent\textbf{Checking Case 1} conditions: $a = 4 \ge 1$,\ $b = 2 > 1$,\ $f(n) = n \ge 0$.

We must verify $f(n) \in O({n^{\log_b a - \varepsilon})}$ for some $\varepsilon > 0$.
Take $\varepsilon = \tfrac{1}{2}$:
\[
    f(n) = n \;\leq\; c \cdot n^{2 - \frac{1}{2}} \;=\; c \cdot n^{\frac{3}{2}}
    \qquad \text{for } c > 1,\ n_0 > 1 \quad\checkmark
\]

Therefore $f(n) \in O{n^2}$, and by the Master Theorem:
\[
    \boxed{T(n) \in \Theta{n^2}}
\]

\bigskip

\subsection*{Question 2}


We apply the \textbf{Master Theorem} with parameters:
\[
    a = 1, \quad b = \tfrac{7}{3}, \quad f(n) = 1, \quad
    n^{\log_b a} = n^{\log_{7/3} 1} = n^0 = 1
\]

\noindent\textbf{Checking Case 2} conditions: $a = 1 \ge 1$,\ $b = \tfrac{7}{3} > 1$,\ $f(n) = 1 \ge 0$.

We must verify $f(n) \in \Theta{n^{\log_b a}}$:
\[
    f(n) = 1 \in \Theta{1} \qquad \text{for } c > 1,\ n_0 > 1 \quad\checkmark
\]

Therefore, by the Master Theorem:
\[
    \boxed{T(n) \in \Theta{\log n}}
\]

\bigskip

\subsection*{Question 3}


We apply the \textbf{Master Theorem} with parameters:
\[
    a = 6, \quad b = 2, \quad f(n) = n^4 \log n, \quad
    n^{\log_b a} = n^{\log_2 6} \approx n^{2.58}
\]

\noindent\textbf{Checking Case 3} conditions: $a = 6 \ge 1$,\ $b = 2 > 1$,\ $f(n) = n^4 \log n \ge 0$.

We must verify $f(n) \in O{n^{\log_b a + \varepsilon}}$ for some $\varepsilon > 0$.
Take $\varepsilon = 1$:
\[
    n^{\log_2 6 + 1} \approx n^{3.58} \;\leq\; n^4 \log n = f(n)
    \qquad \text{for } c = 1,\ n_0 = 1 \quad\checkmark
\]

Now we verify the \textbf{regularity condition}: $a \cdot f\!\left(\tfrac{n}{b}\right) \leq c \cdot f(n)$ for some $c < 1$:
\[
    6 \cdot f\!\left(\tfrac{n}{2}\right) \leq c \cdot f(n)
\]
for $c = \frac{1}{2}$:\\
$6 \cdot \frac{n^4(log(n)-1)}{16} \leq \frac{n^4logn}{2}$
\[
    \frac{3}{8} n^4 (\log n - 1) \leq \frac{1}{2} n^4 \log n
\]
Dividing by $n^4 \log n$ (which is positive for $n > 1$):
\[
    \frac{3}{8} \left( 1 - \frac{1}{\log n} \right) \leq \frac{1}{2}
\]
Since $1 - \frac{1}{\log n} < 1$, the inequality holds for all $n \ge 2$ with $c = \frac{1}{2} < 1$.

\medskip
\noindent\textbf{Conclusion:} 
All conditions of \textbf{Case 3} are satisfied. Therefore:
\[
    \boxed{T(n) = \Theta(f(n)) = \Theta(n^4 \log n)} \quad \checkmark
\]

\subsection*{Question 4}

We apply the \textbf{Master Theorem} with parameters:
\[
    a = 8, \quad b = 2, \quad f(n) = n^4 + 3n^3 \log n, \quad
    n^{\log_b a} = n^{\log_{2} 8} = n^3
\]

\noindent\textbf{Checking Case 3} conditions: $a = 8 \ge 1$,\ $b = 2 > 1$,\ $f(n) = n^4 + 3n^3 \log n \ge 0$.

We must verify $f(n) \in \Omega(n^{3+\epsilon})$ for some $\epsilon > 0$. Take $\epsilon = 1$:
\[
    f(n) = n^4 + 3n^3 \log n \in \Omega(n^{3+1}) = \Omega(n^4) \qquad \checkmark
\]

Now we verify the \textbf{regularity condition}: $a \cdot f(n/b) \leq c \cdot f(n)$ for some $c < 1$ and large $n$:
\[
    8 \cdot f(n/2) = 8 \left( \frac{n^4}{16} + \frac{3n^3}{8} \log\left(\frac{n}{2}\right) \right) = \frac{n^4}{2} + 3n^3(\log n - 1) = \frac{n^4}{2} + 3n^3 \log n - 3n^3
\]
We want to show:
\[
    \frac{n^4}{2} + 3n^3 \log n - 3n^3 \leq c(n^4 + 3n^3 \log n)
\]
For $c = \frac{2}{3}$, the inequality holds since:
\[
    \frac{1}{2}n^4 + 3n^3 \log n - 3n^3 < \frac{2}{3}n^4 + 2n^3 \log n
\]
which is true for sufficiently large $n$.

\medskip
\noindent\textbf{Conclusion:} 
All conditions of \textbf{Case 3} are satisfied. Therefore:
\begin{center}
    $T(n) = \Theta(f(n)) = \Theta(n^4 + 3n^3 \log n)$ 
\end{center}
Since $n^4$ dominates $3n^3\log n$ for large $n$, we have $f(n) = \Theta(n^4)$, and therefore:
\[
    \boxed{T(n) = \Theta(f(n)) = \Theta(n^4)} \quad \checkmark
\]

\subsection*{Question 5}
We will set up the recursion tree.\\
Since $a = 1$, each node has one son.

\begin{itemize}
    \item level 0 (root) - work of $n$
    \item level 1 - work of $\sqrt{n}$
    \item level 2 - work of $\sqrt{\sqrt{n}} = n^{\frac{1}{4}}$
    \item level $i$ - work of $n^{(1/2)^i}$
\end{itemize}

So the total work for $T(n)$ is:
\[
T(n) = \sum_{i=0}^{\log\log n} n^{(1/2)^i}
\]

To prove $T(n) \in \Theta(n)$, we will show $T(n) \in O(n)$ and $T(n) \in \Omega(n)$.

\medskip
\noindent\textbf{First, we prove $T(n) \in O(n)$:}
\[
\sum_{i=0}^{\log\log n} n^{(1/2)^i} 
= n^1 + n^{1/2} + n^{1/4} + \cdots
= n \left(1 + n^{-1/2} + n^{-3/4} + \cdots \right)
\leq n \sum_{i=0}^{\infty} \left(\frac{1}{\sqrt{n}}\right)^i
= n \cdot \frac{1}{1 - n^{-1/2}}
\leq 2n
\]
for $n \geq 4$, since $\frac{1}{1-n^{-1/2}} \leq 2$ when $n^{-1/2} \leq \frac{1}{2}$, i.e.\ $n \geq 4$.
So $T(n) \in O(n)$. $\checkmark$

\medskip
\noindent\textbf{Now, we prove $T(n) \in \Omega(n)$:}
\[
\sum_{i=0}^{\log\log n} n^{(1/2)^i} \geq n^{(1/2)^0} = n
\]
So $T(n) \in \Omega(n)$. $\checkmark$

\noindent Since both bounds hold,\\
\[
    \boxed{T(n) = \Theta(n)} \quad \checkmark
\]
\end{answer}
